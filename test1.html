🔒 Permission Denied — Fix Firestore Rules

Go to Firebase Console → Firestore → Rules and paste:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /{document=**} {
      allow read, write: if true;
    }
  }
}
